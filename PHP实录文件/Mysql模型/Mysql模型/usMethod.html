query($string) array 执行查询语句 $obj->query('SELECT * FROM study');

find($ing) array 执行主键搜索 $obj->find(1);(根据主键搜索,自动查找主键)
where($string) array where语句 $obj->where('id=1')->select();(需要搭配select())
limit($string|$int) array limit语句 $obj->limit(1)->select();(需要搭配select())
order($string) array order by语句 $obj->order('id desc')->select();(需要搭配select());
update($array,$where) int update语句 $obj->query(array("name"=>"admin"),"id=1");
insert($array) int insert语句 $obj->query(array("name"=>"admin"));

          [config.inc.php]里面是MYSQL的配置项
          [functions.php]是作者自己常用的一些小功能(完整的我自己用!@#)
          [Model.class.php]当然就是主要的功能咯!
          [Debug.class.php]如果需要调试就include一下这个文件,里面有个showSql()方法,执行方法时可以看到SQL语句,然而我并没有写CSS
